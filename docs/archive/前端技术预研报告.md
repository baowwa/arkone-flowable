  const samples = ref<Sample[]>([])
  const currentSample = ref<Sample | null>(null)

  async function fetchSamples(params: any) {
    loading.value = true
    try {
      const response = await api.get('/samples', { params })
      samples.value = response.data.records
    } finally {
      loading.value = false
    }
  }

  async function createSample(data: any) {
    const response = await api.post('/samples', data)
    samples.value.unshift(response.data)

    await api.put(`/samples/${id}/status`, { status })
    const sample = samples.value.find(s => s.id === id)
    if (sample) {
      sample.status = status
    }
  }

  return {
    samples,
    currentSample,
    loading,
    inProgressSamples,
    fetchSamples,
    createSample,
    updateSampleStatus
  }
